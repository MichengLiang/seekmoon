= 附录 D：Output Shapes

== `seekmoon search --shape`

[source,text]
----
seekmoon.search-results.v1

schema: string
snapshot:
  id: string
  sources: string[]
query:
  text: string
  kind: library|skill
  target: string|null
results[]:
  rank: int
  module: string
  version: string
  description: evidence<string>
  license: evidence<string>
  repository: evidence<string>
  target: evidence<string[]>
  match:
    fields: string[]
----

== `seekmoon view --shape`

[source,text]
----
seekmoon.module-profile.v1

schema: string
module: string
version: string
manifest:
  latest_version: string
  downloads: int
  build_status: evidence<string>
  metadata: object
docs_url: string
packages:
  count: int
  items[]:
    path: string
    relpath: string
source:
  state: not-fetched|present|failed|unavailable
target:
  status: present|unknown|failed
----
